1.基础操作

// 初始化本地 Git 仓库
git init
->  git init my_project（创建并初始化仓库）

// 查看工作区 / 暂存区状态（新增 / 修改 / 删除）
git status
->  git status（完整状态）、git status -s（简洁状态）

// 将文件添加到暂存区
git add <文件/目录>	
git add .（添加所有更改）
git add src/App.vue（添加指定文件）

// 提交暂存区更改到本地仓库
git commit -m "备注"
->  git commit -m "完成用户登录功能"

// 查看提交历史
git log
->  git log --oneline（精简一行显示）
->  git log --graph（图形化分支）


2.仓库管理（关联 / 同步远程）

// 克隆远程仓库到本地
git clone <远程地址>
->  git clone git@github.com:xxx/remember_app.git

// 查看本地关联的远程仓库（名称 + 地址）
git remote -v
->  git remote -v（输出 origin）

// 添加远程仓库
git remote add <远程名> <地址>
->  git remote add origin git@github.com:xxx/backend.git

// 删除指定远程仓库
git remote remove <远程名>
->  git remote remove origin

// 拉取远程代码并合并到本地
git pull <远程名> <分支>
->  git pull origin main
->  git pull origin main --allow-unrelated-histories（合并不相关历史）

// 推送本地提交到远程仓库
git push <远程名> <分支>
->  git push origin main
->  git push -f origin main（强制推送，慎用）


3.分支操作（多人协作核心）

// 查看本地分支（* 标记当前分支）
git branch

// 查看所有分支（本地 + 远程）
git branch -a

// 创建新分支
git branch <分支名>
->  git branch dev（创建 dev 分支）

// 切换分支
git checkout <分支名>
->  git checkout dev
->  git checkout -b feature/login（创建并切换分支）

// 合并指定分支到当前分支
git merge <分支名>
->  git merge dev（将 dev 合并到 main）

// 删除本地分支
git branch -d <分支名>
->  git branch -d dev（删除已合并的分支）


4.文件操作（删除 / 恢复 / 忽略）

// 删除文件并加入暂存区
git rm <文件/目录>
->  git rm -rf folder（删除文件夹）

// 恢复未暂存的文件（撤销修改）
git checkout -- <文件>
->  git checkout -- src/App.vue

// 将文件从暂存区撤回工作区
git restore --staged <文件>
->  git restore --staged .gitignore

// 创建.gitignore（忽略指定文件）
echo "node_modules/" > .gitignore (忽略 node_modules、venv 等无需提交的文件)


5.历史 / 回滚（排错 / 撤销）

// 回滚提交（保留本地修改）
git reset --soft <提交哈希>
->  git reset --soft 1234567（回滚到指定版本）

// 强制回滚（丢弃本地修改，慎用）
git reset --hard <提交哈希>
->  git reset --hard origin/main（同步远程最新版本）

// 暂存未提交的修改（临时切换分支用）
git stash
->  git stash（暂存）
->  git stash pop（恢复暂存）

// 交互式变基（整理 / 删除提交）
git rebase -i <提交范围>
->  git rebase -i HEAD~5（编辑最近 5 条提交）


6.冲突解决（合并 / 拉取冲突）

// 放弃合并，回到合并前状态
git merge --abort（合并冲突时执行，撤销合并操作）

// 放弃变基，回到变基前状态
git rebase --abort(变基冲突时执行)

// 查看文件冲突内容
git diff
->  git diff（所有冲突）
->  git diff <文件>（指定文件）


// 首次推送到 GitHub 新建的空仓库流程
git init    // 1. 本地初始化 Git 仓库（如果未初始化）
git remote add origin git@github.com:xxx/your-repo.git  // 2. 关联远程仓库（替换为你的仓库地址）
git add .   // 3. 暂存+提交本地代码
git commit -m "初始化项目：提交基础代码"
git push -u origin main     // 4. 首次推送（-u 关联分支，后续可直接 git push）


// 本地需同时推送到多个远程仓库（如 frontend/backend 分开推送）流程
git remote -v   // 1. 查看已关联的远程仓库
git remote add frontend git@github.com:xxx/remember_app_frontend.git    // 2. 添加多个远程仓库（命名区分）
git remote add backend git@github.com:xxx/remember_app_backend.git
git push frontend main  // 推送到前端仓库    // 3. 推送到指定仓库（核心：指定远程仓库名）
git push backend main   // 推送到后端仓库
git pull frontend main  // 4. 拉取指定仓库的最新代码
git pull backend main

// 清理远程仓库错误代码流程
git clone 仓库地址  // 0.先确保本地有两个仓库的完整克隆（如果没有，先克隆，有就跳过）
cd /path/to/your/directory  // 1.进入本地仓库目录
git rm -rf [--cached] files    // 2. 删除相关文件/文件夹(多个文件空格隔开)      --cached 参数（只从 Git 仓库中删除，保留本地文件）。
git rm -rf frontend/  // 如果有整段的前端文件夹，直接删
git add .   // 3. 提交删除操作
git commit -m "错误提示"
git push origin main    // 4. 推送到远程仓库（完成远程清理）


// 本地与远程仓库同步（日常开发闭环）流程
git pull origin main   // 1. 拉取远程最新代码（优先拉取，避免推送冲突） 强制提交两个“无关联”的提交历史 --allow-unrelated-histories
git add .   // 2. 开发后暂存+提交本地更改
git commit -m "完成xxx功能开发"
git push origin main    // 3. 推送本地代码到远程

// （可选）如果拉取/推送失败，解决冲突后重试
// 查看冲突文件：git status
// 手动解决冲突后：git add . → git commit → git push


// 推送失败、提交错误代码，需回滚到正确版本流程
// 场景A：推送失败（网络/权限问题），先暂存本地修改
git stash  // 暂存未提交的修改
git pull origin main  // 拉取远程最新
git stash pop  // 恢复暂存的修改，解决冲突后重新推送

// 场景B：错误提交已推送到远程，强制回滚（慎用）
git log --oneline  // 找到正确版本的哈希值（如 1234567）
git reset --hard 1234567  // 本地回滚到正确版本
git push -f origin main  // 强制推送到远程（覆盖错误提交）
